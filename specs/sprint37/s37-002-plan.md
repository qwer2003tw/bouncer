def _build_resource_summary(resource_changes: list, max_items: int = 10) -> list:
    """Extract minimal fields from CFN Changes[] for SFN payload.
    
    Returns list of dicts with: logical_id, action, resource_type, replacement
    max_items 防止 SFN payload 過大。
    """
    summary = []
    for change in resource_changes[:max_items]:
        rc = change.get('ResourceChange', {})
        summary.append({
            'logical_id': rc.get('LogicalResourceId', '?'),
            'action': rc.get('Action', '?'),
            'resource_type': rc.get('ResourceType', '?'),
            'replacement': rc.get('Replacement', 'False'),
        })
    return summary

#### 1b. `handle_infra_approval_request()` — 格式化明細

```python
def _format_change_detail(summary: list, has_more: bool, total: int) -> str:
    """Format resource_changes_summary into Telegram-ready text block."""
    if not summary:
        return ''
    
    ACTION_EMOJI = {
        'Add': '➕',
        'Modify': '✏️',
        'Delete': '🗑️',
    }
    REPLACEMENT_FLAG = {
        'True': ' ⚠️ 需替換',
        'Conditional': ' ⚠️ 可能替換',
        'False': '',
    }
    
    lines = ['📋 *變更明細：*']
    for item in summary:
        emoji = ACTION_EMOJI.get(item['action'], '❓')
        lid = item['logical_id'][:40]
        rtype = item['resource_type']
        repl = REPLACEMENT_FLAG.get(item['replacement'], '')
        lines.append(f"{emoji} `{lid}` ({rtype}){repl}")
    
    if has_more:
        shown = len(summary)
        remaining = total - shown
        lines.append(f"\\.\\.\\. 及 {remaining} 個其他變更")
    
    return '\n'.join(lines)
```
